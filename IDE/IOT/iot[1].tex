\begin{multicols}{2}
\begin{enumerate}[label=(\Alph*)]
  \item $A + B$
  \item $A \cdot \bar{B}$
  \item $A + \bar{B}$
  \item $\bar{B}$
\end{enumerate}
\end{multicols}

\section{Answer}
The above question can be solved as follows:\\
$\rightarrow \bar A \cdot \bar B + A \cdot \bar B$\\
$\rightarrow (A + \bar A )\cdot \bar{B}$\\
$\rightarrow \bar{B}$\\
Therefore, the output $F=\bar B$.

\section{K-Map Implementation}
\begin{center}
\begin{karnaugh-map}[2][2][1][$B$][$A$]
  \maxterms{1,3}
  \minterms{0,2}
  \implicant{0}{0}
  \implicant{2}{2}
\end{karnaugh-map}
\end{center}
Therefore $F = \bar{B}$.
